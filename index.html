How to make this work?

Browsers cannot directly become USB devices. You need a small script on your PC to receive these commands and simulate a controller.

import asyncio
import websockets
import json
import vgamepad as vg # pip install vgamepad

pad = vg.VX360Gamepad()

async def handler(websocket):
    print("Controller Connected")
    async for message in websocket:
        data = json.loads(message)
        if data['type'] == 'input':
            # Joysticks
            pad.left_joystick_float(x_value_float=data['joysticks']['left']['x'], y_value_float=-data['joysticks']['left']['y'])
            pad.right_joystick_float(x_value_float=data['joysticks']['right']['x'], y_value_float=-data['joysticks']['right']['y'])
            
            # Buttons (Example for A/B/X/Y)
            btns = data['buttons']
            if btns.get('A'): pad.press_button(button=vg.XUSB_BUTTON.XUSB_GAMEPAD_A)
            else: pad.release_button(button=vg.XUSB_BUTTON.XUSB_GAMEPAD_A)
            
            # ... Add other buttons mapping here ...
            
            pad.update()

async def main():
    async with websockets.serve(handler, "0.0.0.0", 8080):
        print("Server running on port 8080...")
        await asyncio.Future()  # run forever

if __name__ == "__main__":
    asyncio.run(main())

Requires Python installed. Run: pip install vgamepad websockets
